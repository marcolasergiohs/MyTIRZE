2. Substitua o código de gravação do seu `Código.gs` pelo código abaixo:

function doGet() {
  return HtmlService.createHtmlOutputFromFile('Index')
      .setTitle('Mounjaro Clínico Pro')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function salvarDados(dados) {
  var idPlanilha = SpreadsheetApp.getActiveSpreadsheet().getId();
  var planilha = SpreadsheetApp.openById(idPlanilha).getSheetByName("Registros");
  
  planilha.appendRow([
    dados.data,
    dados.peso,
    dados.dose,
    dados.local,
    dados.agua,
    dados.proteina,
    dados.intestino,
    "Nausea: " + dados.nausea + ", Azia: " + dados.azia + ", Fadiga: " + dados.fadiga + ", Glicemia: " + dados.glicemia,
    dados.notas
  ]);
  
  return "Registro gravado com sucesso no Google Planilhas!";
}
